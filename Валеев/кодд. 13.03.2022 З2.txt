#include <stdio.h>
#include <stdlib.h>
#include <time.h>

int main()
{
    srand(time(NULL));
    int *M,n,i_max=0,y;
    printf("Введите число\n");
    scanf("%d",&y);
    printf("Введите кол. элементов\n");
    scanf("%d",&n);
    M=new int[n];
    for(int i=0; i<n; i++)
    {
        M[i] = rand()%50;
        printf("%d ", M[i]);
    }
    int max=M[0];
    for(int i=0;i<n;i++)
    {
        if(M[i]>max)
        {
            max=M[i];
        }
    }
    for(int i=0;i<n;i++)
    {
        if(M[i]>y)
        {
            M[i]=max*2;
        }
    }
    printf("\n");
    for(int i=0; i<n; i++)
        printf("%d ", M[i]);
    delete []M;
    return 0;
}
